Problem: Given an array of integers, implement Merge Sort to sort the array efficiently.

Time Complexity: Best: O(n log n) Average: O(n log n) Worst: O(n log n)

// Merge Sort Implementation (Long + Optimized + Detailed Comments)
// -----------------------------------------------------------------

// Function to merge two sorted halves
function merge(left, right) {
    let result = [];
    let i = 0, j = 0;

    // Compare elements of left and right arrays
    while (i < left.length && j < right.length) {
        if (left[i] < right[j]) {
            result.push(left[i]);
            i++;
        } else {
            result.push(right[j]);
            j++;
        }
    }

    // Add remaining elements (if any)
    return result.concat(left.slice(i)).concat(right.slice(j));
}

// Main Merge Sort function
function mergeSort(arr) {
    // Base case: single element
    if (arr.length <= 1) return arr;

    // Split array into halves
    const mid = Math.floor(arr.length / 2);
    const left = arr.slice(0, mid);
    const right = arr.slice(mid);

    // Recursively sort halves and merge
    return merge(mergeSort(left), mergeSort(right));
}

// Driver Code
let array = [64, 25, 12, 22, 11, 90, 34, 77, 55, 3];
console.log("Original Array:", array);

let sortedArray = mergeSort(array);
console.log("Sorted Array:", sortedArray);
