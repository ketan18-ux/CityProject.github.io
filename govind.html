Problem & Solution
Optimize city project selection under limited budget using Knapsack DP.
C++ Code
#include <bits/stdc++.h> using namespace std; int main(){ int n,W;cin>>n>>W; vector<int> w(n),v(n); for(int i=0;i<n;i++) cin>>w[i]>>v[i]; vector<long long> dp(W+1,0); for(int i=0;i<n;i++) for(int c=W;c>=w[i];c--) dp[c]=max(dp[c],dp[c-w[i]]+v[i]); cout<<dp[W]; }

Metro Construction Phase Optimization
Heap + Heap Sort SDG 9 – Industry, Innovation and Infrastructure SDG 11 – Sustainable Cities and Communities SDG 8 – Decent Work and Economic Growth
Prioritize metro phases based on ROI.
C++ Code
#include <bits/stdc++.h> using namespace std; void heapify(vector<int>& a,int n,int i){ int largest=i,l=2*i+1,r=2*i+2; if(l<n&&a[l]>a[largest]) largest=l; if(r<n&&a[r]>a[largest]) largest=r; if(largest!=i){ swap(a[i],a[largest]); heapify(a,n,largest); } } void heapSort(vector<int>& a){ int n=(int)a.size(); for(int i=n/2-1;i>=0;i--) heapify(a,n,i); for(int i=n-1;i>=0;i--){ swap(a[0],a[i]); heapify(a,i,0); } } int main(){ int n;cin>>n; vector<int> a(n); for(auto &x:a) cin>>x; heapSort(a); for(auto &x:a) cout<<x<<' '; }

Subsidy Distribution Optimization
QuickSort + Dynamic Programming SDG 1 – No Poverty SDG 3 – Good Health and Well-Being
Optimize subsidy allocation using sorting and DP.
C++ Code
#include <bits/stdc++.h> using namespace std; void quicksort(vector<int>& a,int l,int r){ if(l<r){ int pivot=a[r],i=l-1; for(int j=l;j<r;j++){ if(a[j]<pivot){ i++;swap(a[i],a[j]); } } swap(a[i+1],a[r]); quicksort(a,l,i); quicksort(a,i+2,r); } } int main(){ int n;cin>>n; vector<int> subs(n); for(auto &x:subs) cin>>x; quicksort(subs,0,n-1); // DP logic here cout<<"Optimized subsidy distribution\n"; }
