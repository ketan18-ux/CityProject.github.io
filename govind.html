City Budget Allocation
SDG 1 – No Poverty SDG 11 – Sustainable Cities and Communities SDG 8 – Decent Work and Economic Growth
Problem
A city has limited budget for multiple development projects. Each project has a cost and expected benefit. Determine which projects to fund to maximize total benefit without exceeding the budget.
Solution
Use 0/1 Knapsack Dynamic Programming to select projects with maximum benefit while respecting budget constraint.
C++ Code
#include <bits/stdc++.h> using namespace std; int main(){ int n,W; cin>>n>>W; vector<int> w(n),v(n); for(int i=0;i<n;i++) cin>>w[i]>>v[i]; vector<long long> dp(W+1,0); for(int i=0;i<n;i++) for(int c=W;c>=w[i];c--) dp[c]=max(dp[c],dp[c-w[i]]+v[i]); cout<<dp[W]; }
Tax Officer Routing
SDG 8 – Decent Work and Economic Growth SDG 9 – Industry, Innovation and Infrastructure SDG 11 – Sustainable Cities and Communities
Problem
Optimize routes for tax officers to minimize travel time between multiple locations. Each road has a travel cost.
Solution
Apply Dijkstra's shortest path algorithm to compute minimal travel cost between points.
C++ Code
#include <bits/stdc++.h> using namespace std; int main(){ int n,m; cin>>n>>m; vector<vector<pair<int,int>>> g(n); for(int i=0;i<m;i++){ int u,v,w; cin>>u>>v>>w; g[u].push_back({v,w}); g[v].push_back({u,w}); } vector<int> dist(n,1e9); priority_queue<pair<int,int>,vector<pair<int,int>>,greater<>> pq; dist[0]=0; pq.push({0,0}); while(!pq.empty()){ auto[d,u]=pq.top(); pq.pop(); if(d!=dist[u]) continue; for(auto &e:g[u]){ auto[v,w]=e; if(dist[v]>d+w){ dist[v]=d+w; pq.push({dist[v],v}); } } } cout<<dist[n-1]; }
Smart Parking Fee
SDG 11 – Sustainable Cities and Communities SDG 9 – Industry, Innovation and Infrastructure SDG 8 – Decent Work and Economic Growth
Problem
Calculate dynamic parking fees based on hourly occupancy to optimize city parking revenue.
Solution
Use segment tree to efficiently query and update hourly occupancy data to determine optimal fee.
C++ Code
#include <bits/stdc++.h> using namespace std; struct Seg{ int n; vector<long long> st; Seg(int n):n(n),st(4*n,0){} long long build(vector<long long>&a,int p,int l,int r){ if(l==r) return st[p]=a[l]; int m=(l+r)/2; return st[p]=max(build(a,2*p,l,m),build(a,2*p+1,m+1,r)); } long long qry(int p,int l,int r,int ql,int qr){ if(qr<l||ql>r) return -1e18; if(ql<=l&&r<=qr) return st[p]; int m=(l+r)/2; return max(qry(2*p,l,m,ql,qr),qry(2*p+1,m+1,r,ql,qr)); } }; int main(){ int n; cin>>n; vector<long long> a(n); for(long long &x:a) cin>>x; Seg seg(n); seg.build(a,1,0,n-1); int q; cin>>q; while(q--){ int l,r; cin>>l>>r; cout<<seg.qry(1,0,n-1,l,r)<<"\n"; } }
Pipeline Leak Detection
SDG 6 – Clean Water and Sanitation SDG 13 – Climate Action SDG 11 – Sustainable Cities and Communities
Problem
Detect potential leaks in pipeline network based on pressure and pipeline connectivity.
Solution
Apply DFS traversal to simulate pressure propagation and detect low-pressure nodes indicating leaks.
C++ Code
#include <bits/stdc++.h> using namespace std; vector<vector<int>> g; vector<int> leak; void dfs(int u,int p,int pressure){ if(pressure<0) leak[u]=1; for(int v:g[u]) if(v!=p) dfs(v,u,pressure-1); } int main(){ int n; cin>>n; g.assign(n,{}); for(int i=0;i<n-1;i++){ int u,v; cin>>u>>v; g[u].push_back(v); g[v].push_back(u); } leak.assign(n,0); dfs(0,-1,10); for(int i=0;i<n;i++) if(leak[i]) cout<<"Leak at: "<<i<<"\n"; }
