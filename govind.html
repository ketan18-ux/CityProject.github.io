City Budget Allocation
SDG 1 – No Poverty SDG 11 – Sustainable Cities and Communities SDG 8 – Decent Work and Economic Growth
Problem
Optimize city project selection under limited budget using Knapsack DP.
Solution
Use 0/1 Knapsack Dynamic Programming to select projects maximizing total benefit without exceeding budget.
C++ Code
#include <bits/stdc++.h> using namespace std; int main(){ int n,W; cin>>n>>W; vector<int> w(n),v(n); for(int i=0;i<n;i++) cin>>w[i]>>v[i]; vector<long long> dp(W+1,0); for(int i=0;i<n;i++) for(int c=W;c>=w[i];c--) dp[c]=max(dp[c],dp[c-w[i]]+v[i]); cout<<dp[W]; }

Smart Parking Fee
SDG 11 – Sustainable Cities and Communities SDG 9 – Industry, Innovation and Infrastructure SDG 8 – Decent Work and Economic Growth
Problem
Calculate dynamic parking fees based on hourly occupancy to maximize revenue.
Solution
Use segment trees to efficiently query and update occupancy data to set optimal fees.
C++ Code
#include <bits/stdc++.h> using namespace std; struct Seg{ int n; vector<long long> st; Seg(int n):n(n),st(4*n,0){} long long build(vector<long long>&a,int p,int l,int r){ if(l==r) return st[p]=a[l]; int m=(l+r)/2; return st[p]=max(build(a,2*p,l,m),build(a,2*p+1,m+1,r)); } long long qry(int p,int l,int r,int ql,int qr){ if(qr<l||ql>r) return -1e18; if(ql<=l&&r<=qr) return st[p]; int m=(l+r)/2; return max(qry(2*p,l,m,ql,qr),qry(2*p+1,m+1,r,ql,qr)); } }; int main(){ int n; cin>>n; vector<long long> a(n); for(long long &x:a) cin>>x; Seg seg(n); seg.build(a,1,0,n-1); int q; cin>>q; while(q--){ int l,r; cin>>l>>r; cout<<seg.qry(1,0,n-1,l,r)<<"\n"; } }
Pipeline Leak Detection
SDG 6 – Clean Water and Sanitation SDG 13 – Climate Action SDG 11 – Sustainable Cities and Communities
Problem
Detect leaks in a city pipeline network based on pressure measurements.
Solution
Use DFS to simulate pressure propagation and identify nodes with low pressure indicating leaks.
C++ Code
#include <bits/stdc++.h> using namespace std; vector<vector<int>> g; vector<int> leak; void dfs(int u,int p,int pressure){ if(pressure<0) leak[u]=1; for(int v:g[u]) if(v!=p) dfs(v,u,pressure-1); } int main(){ int n; cin>>n; g.assign(n,{}); for(int i=0;i<n-1;i++){ int u,v; cin>>u>>v; g[u].push_back(v); g[v].push_back(u); } leak.assign(n,0); dfs(0,-1,10); for(int i=0;i<n;i++) if(leak[i]) cout<<"Leak at: "<<i<<"\n"; }
Metro Phase Optimization
SDG 9 – Industry, Innovation and Infrastructure SDG 11 – Sustainable Cities and Communities
Problem
Optimize metro line construction phases to minimize total cost and maximize coverage.
Solution
Use Minimum Spanning Tree algorithms (Prim/Kruskal) to select construction sequence with minimal cost.
C++ Code
#include <bits/stdc++.h> using namespace std; struct Edge{ int u,v,w; }; int find(vector<int>&p,int x){ return p[x]==x?x:p[x]=find(p,p[x]); } int main(){ int n,m; cin>>n>>m; vector<Edge> e(m); for(int i=0;i<m;i++) cin>>e[i].u>>e[i].v>>e[i].w; sort(e.begin(),e.end(),[](Edge &a,Edge &b){return a.w<b.w;}); vector<int>p(n); iota(p.begin(),p.end(),0); long long cost=0; for(auto &x:e){ int a=find(p,x.u),b=find(p,x.v); if(a!=b){p[a]=b; cost+=x.w;} } cout<<cost; }
