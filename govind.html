1. City Budget Allocation
Algorithm: 0/1 Knapsack (Dynamic Programming)

Problem: Select the best set of city projects within a limited budget.

Solution: Use DP to evaluate all possible project combinations and compute maximum possible impact under the given budget.

SDG 1: No Poverty SDG 8: Economic Growth SDG 11: Sustainable Cities
C++ Code
#include <bits/stdc++.h> using namespace std; struct Project { int cost; int impact; string name; }; int main() { int n, W; cin >> n >> W; vector p(n); for (int i = 0; i < n; i++) cin >> p[i].name >> p[i].cost >> p[i].impact; vector> dp(n + 1, vector(W + 1, 0)); for (int i = 1; i <= n; i++) { for (int cap = 0; cap <= W; cap++) { if (p[i - 1].cost <= cap) dp[i][cap] = max(dp[i - 1][cap], p[i - 1].impact + dp[i - 1][cap - p[i - 1].cost]); else dp[i][cap] = dp[i - 1][cap]; } } cout << "Maximum Impact = " << dp[n][W]; }

3. Smart Parking Fee Prediction
Algorithm: Segment Tree + Range Sum

Problem: The city wants to predict parking fees dynamically based on usage patterns over time.

Solution: A Segment Tree efficiently computes range usage data. Fee is calculated dynamically based on intensity of parking demand in queried intervals.

SDG 9: Industry & Innovation SDG 11: Sustainable Cities SDG 13: Climate Action
C++ Code
#include <bits/stdc++.h> using namespace std; struct SegmentTree { int n; vector tree; SegmentTree(int n) : n(n), tree(4 * n, 0) {} void build(vector& arr, int node, int start, int end) { if (start == end) { tree[node] = arr[start]; } else { int mid = (start + end) / 2; build(arr, 2 * node, start, mid); build(arr, 2 * node + 1, mid + 1, end); tree[node] = tree[2 * node] + tree[2 * node + 1]; } } long long query(int node, int start, int end, int l, int r) { if (r < start || end < l) return 0; if (l <= start && end <= r) return tree[node]; int mid = (start + end) / 2; long long left = query(2 * node, start, mid, l, r); long long right = query(2 * node + 1, mid + 1, end, l, r); return left + right; } }; int main() { int n; cin >> n; vector usage(n); for (int i = 0; i < n; i++) cin >> usage[i]; SegmentTree seg(n); seg.build(usage, 1, 0, n - 1); int q; cin >> q; while (q--) { int L, R; cin >> L >> R; long long demand = seg.query(1, 0, n - 1, L, R); int fee; if (demand <= 200) fee = 10; else if (demand <= 500) fee = 20; else if (demand <= 1000) fee = 30; else fee = 50; cout << fee << "\n"; } }
4. Public Transport Scheduling
Algorithm: Fenwick Tree + Moving Average Forecast

Problem: Forecast daily travel demand to generate efficient bus schedules.

Solution: Fenwick Tree supports fast prefix sums. A rolling-average recurrence predicts future passenger counts.

SDG 7: Clean Energy SDG 11: Sustainable Cities SDG 13: Climate Action
C++ Code
#include <bits/stdc++.h> using namespace std; struct Fenwick { int n; vector bit; Fenwick(int n) : n(n), bit(n + 1, 0) {} void update(int i, long long val) { for (++i; i <= n; i += i & -i) bit[i] += val; } long long sum(int i) { long long s = 0; for (++i; i > 0; i -= i & -i) s += bit[i]; return s; } long long range(int l, int r) { return sum(r) - (l ? sum(l - 1) : 0); } }; int main() { int n; cin >> n; vector usage(n); for (int i = 0; i < n; i++) cin >> usage[i]; Fenwick fen(n); for (int i = 0; i < n; i++) fen.update(i, usage[i]); int k; cin >> k; int window = 5; vector forecast; for (int t = 0; t < k; t++) { int start = max(0, n - window); long long total = fen.range(start, n - 1); long long next = total / (min(window, n)); forecast.push_back(next); usage.push_back(next); fen.update(n, next); n++; } for (auto x : forecast) cout << x << " "; }

6. Streetlight Energy Reduction
Algorithm: Backtracking + Scheduling

Problem: Reduce electricity consumption while maintaining minimum safe brightness.

Solution: Backtracking explores all safe brightness combinations per hour; minimum-cost schedule is selected.

SDG 7: Clean Energy SDG 11: Sustainable Cities SDG 13: Climate Action
C++ Code
#include <bits/stdc++.h> using namespace std; int H, minBrightness; vector level; vector bestSchedule; long long bestCost = LLONG_MAX; void backtrack(int hour) { if (hour == H) { long long cost = accumulate(level.begin(), level.end(), 0LL); if (cost < bestCost) { bestCost = cost; bestSchedule = level; } return; } for (int b = minBrightness; b <= 100; b += 10) { level[hour] = b; backtrack(hour + 1); } } int main() { cin >> H >> minBrightness; level.assign(H, 0); backtrack(0); cout << "Minimum Cost: " << bestCost << "\n"; for (int x : bestSchedule) cout << x << " "; }

8. Garbage Plant Load Balancing
Algorithm: Min Heap + Queue Simulation

Problem: Evenly distribute waste across plants to avoid overload.

Solution: Use a min-heap to always assign waste to the plant with the smallest current load.

SDG 11: Sustainable Cities SDG 12: Responsible Consumption SDG 13: Climate Action
C++ Code
#include <bits/stdc++.h> using namespace std; struct Plant { long long load; int id; bool operator<(const Plant &other) const { return load > other.load; } }; int main() { int plants, arrivals; cin >> plants >> arrivals; priority_queue pq; for (int i = 0; i < plants; i++) pq.push({0, i}); for (int i = 0; i < arrivals; i++) { long long w; cin >> w; auto plant = pq.top(); pq.pop(); plant.load += w; cout << "Assigned " << w << " to plant " << plant.id << "\n"; pq.push(plant); } cout << "Final Loads:\n"; vector loads(plants); while (!pq.empty()) { auto p = pq.top(); pq.pop(); loads[p.id] = p.load; } for (int i = 0; i < plants; i++) cout << "Plant " << i << ": " << loads[i] << "\n"; }
9. Building Maintenance Prediction
Algorithm: 2-3 Tree (Multiset) + Divide & Conquer

Problem: Predict recurring maintenance patterns in buildings.

Solution: Store logs in a multiset. Divide-and-conquer detects repetition patterns by comparing mid boundaries.

SDG 9: Infrastructure SDG 11: Sustainable Cities SDG 12: Responsible Consumption
C++ Code
#include <bits/stdc++.h> using namespace std; int main() { int n; cin >> n; multiset logs; for (int i = 0; i < n; i++) { int t; cin >> t; logs.insert(t); } vector arr(logs.begin(), logs.end()); function solve = [&](int l, int r) { if (l == r) return 1; int mid = (l + r) / 2; int left = solve(l, mid); int right = solve(mid + 1, r); if (arr[mid] == arr[mid + 1]) return left + right - 1; return left + right; }; cout << "Pattern Score = " << solve(0, arr.size() - 1); }
10. Subsidy Distribution Optimization
Algorithm: Quick Sort + 0/1 Knapsack

Problem: Select beneficiaries who maximize social impact under a limited subsidy fund.

Solution: Sort applicants by benefit-to-cost ratio, then apply 0/1 knapsack to choose optimal beneficiaries.

SDG 1: No Poverty SDG 8: Economic Growth SDG 10: Reduced Inequalities
C++ Code
#include <bits/stdc++.h> using namespace std; struct Beneficiary { double benefit, cost; }; int main() { int n, W; cin >> n >> W; vector b(n); for (int i = 0; i < n; i++) cin >> b[i].benefit >> b[i].cost; sort(b.begin(), b.end(), [&](auto &A, auto &B) { return (A.benefit / A.cost) > (B.benefit / B.cost); }); vector w(n), v(n); for (int i = 0; i < n; i++) { w[i] = round(b[i].cost); v[i] = round(b[i].benefit); } vector dp(W + 1, 0); for (int i = 0; i < n; i++) { for (int cap = W; cap >= w[i]; cap--) { dp[cap] = max(dp[cap], dp[cap - w[i]] + v[i]); } } cout << "Max Benefit = " << *max_element(dp.begin(), dp.end()); }
