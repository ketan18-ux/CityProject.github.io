Problem 1: Best Allocation of City Budget


#include <bits/stdc++.h>
using namespace std;
void problem1() {
    int n, budget;
    cout << "Problem 1: Best Allocation of City Budget\n";
    cout << "Enter number of projects and total budget: ";
    cin >> n >> budget;
    vector<int> cost(n), impact(n);
    cout << "Enter cost and impact of each project:\n";
    for(int i = 0; i < n; i++) cin >> cost[i] >> impact[i];
    vector<int> dp(budget+1,0);
    for(int i=0;i<n;i++){
        for(int j=budget;j>=cost[i];j--){
            dp[j]=max(dp[j],dp[j-cost[i]]+impact[i]);
        }
    }
    cout << "Maximum impact achievable: " << *max_element(dp.begin(),dp.end()) << "\n\n";
}

Problem 2: Tax Officer Route Optimization


#include <bits/stdc++.h>
using namespace std;
void problem2() {
    int n,m;
    cout << "Problem 2: Tax Officer Route Optimization\n";
    cout << "Enter number of zones and roads: ";
    cin >> n >> m;
    vector<vector<pair<int,int>>> adj(n);
    cout << "Enter roads (u v w):\n";
    for(int i=0;i<m;i++){
        int u,v,w; cin >> u >> v >> w;
        adj[u].push_back({v,w});
        adj[v].push_back({u,w});
    }
    int start;
    cout << "Enter starting zone: ";
    cin >> start;
    const long long INF=1e18;
    vector<long long> dist(n,INF);
    dist[start]=0;
    priority_queue<pair<long long,int>, vector<pair<long long,int>>, greater<>> pq;
    pq.push({0,start});
    while(!pq.empty()){
        auto [d,u]=pq.top(); pq.pop();
        if(d!=dist[u]) continue;
        for(auto [v,w]: adj[u]){
            if(dist[v]>d+w){
                dist[v]=d+w;
                pq.push({dist[v],v});
            }
        }
    }
    cout << "Shortest distances from start:\n";
    for(auto d: dist) cout << d << " ";
    cout << "\n\n";
}

Problem 3: Smart Parking Fee Prediction


#include <bits/stdc++.h>
using namespace std;
struct SegmentTree {
    vector<long long> tree;
    int n;
    SegmentTree(int size){ n=size; tree.assign(4*n,0);}
    void build(vector<int> &a,int v,int tl,int tr){
        if(tl==tr) tree[v]=a[tl];
        else {
            int tm=(tl+tr)/2;
            build(a,2*v,tl,tm);
            build(a,2*v+1,tm+1,tr);
            tree[v]=tree[2*v]+tree[2*v+1];
        }
    }
    long long query(int v,int tl,int tr,int l,int r){
        if(l>r) return 0;
        if(l==tl && r==tr) return tree[v];
        int tm=(tl+tr)/2;
        return query(2*v,tl,tm,l,min(r,tm))+query(2*v+1,tm+1,tr,max(l,tm+1),r);
    }
};
void problem3() {
    int n,q; cout << "Problem 3: Smart Parking Fee Prediction\n";
    cout << "Enter number of parking slots: "; cin >> n;
    vector<int> usage(n); cout << "Enter parking usage:\n";
    for(int i=0;i<n;i++) cin >> usage[i];
    SegmentTree seg(n); seg.build(usage,1,0,n-1);
    cout << "Enter number of queries: "; cin >> q;
    while(q--){
        int l,r; cin >> l >> r;
        long long demand=seg.query(1,0,n-1,l,r);
        int fee = demand<=100?10:demand<=500?20:demand<=1000?30:50;
        cout << "Predicted fee: " << fee << "\n";
    }
    cout << "\n";
}

Problem 4: City Energy Demand Forecasting


#include <bits/stdc++.h>
using namespace std;
void problem4() {
    int n; cout << "Problem 4: City Energy Demand Forecasting\n";
    cout << "Enter number of past days: "; cin >> n;
    vector<double> x(n),y(n);
    cout << "Enter day index and energy consumption:\n";
    for(int i=0;i<n;i++) cin >> x[i] >> y[i];
    double sumX=0,sumY=0,sumXY=0,sumXX=0;
    for(int i=0;i<n;i++){ sumX+=x[i]; sumY+=y[i]; sumXY+=x[i]*y[i]; sumXX+=x[i]*x[i]; }
    double slope=(n*sumXY - sumX*sumY)/(n*sumXX - sumX*sumX);
    double intercept=(sumY - slope*sumX)/n;
    double nextDay=x.back()+1;
    double prediction=slope*nextDay + intercept;
    cout << "Predicted energy: " << prediction << "\n\n";
}

Problem 5: Metro Construction Phase Optimization


#include <bits/stdc++.h>
using namespace std;
void problem5() {
    int n,m; cout << "Problem 5: Metro Construction Phase Optimization\n";
    cout << "Enter number of stations and phases: "; cin >> n >> m;
    vector<int> cost(n); cout << "Enter construction cost per station:\n";
    for(int i=0;i<n;i++) cin >> cost[i];
    vector<int> dp(n+1,INT_MAX/2);
    dp[0]=0;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=min(i,m);j++){
            dp[i]=min(dp[i],dp[i-j]+cost[i-1]);
        }
    }
    cout << "Minimum construction cost: " << dp[n] << "\n\n";
}

Problem 8: Garbage Plant Load Balancing


#include <bits/stdc++.h>

    vector<int> load(n); cout << "Enter initial load:\n";
    for(int i=0;i<n;i++) cin >> load[i];

Problem 10: Subsidy Distribution Optimization


#include <bits/stdc++.h>
using namespace std;
void problem10() {
    int n,budget; cout << "Problem 10: Subsidy Distribution Optimization\n";
    cout << "Enter number of citizens and total subsidy budget: "; cin >> n >> budget;
    vector<int> need(n); cout << "Enter subsidy need per citizen:\n";
    for(int i=0;i<n;i++) cin >> need[i];
    sort(need.begin(),need.end());
    int count=0;
    for(int i=0;i<n;i++){
        if(budget>=need[i]){ budget-=need[i]; count++; } else break;
    }
    cout << "Maximum citizens receiving subsidy: " << count << "\n\n";
}
